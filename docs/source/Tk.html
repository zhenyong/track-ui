'use strict';
/**
 * @class Tk
 */
(function(global) {

	var Tk = global.Tk = global.Tk || {global:global}; //全局命名空间

	var objectPrototype = Object.prototype,
		toString = objectPrototype.toString,
		enumerables = ['valueOf', 'toLocaleString', 'toString', 'constructor'], i;
	
	/**
	 * 参考 ref [jquery#extend](http://api.jquery.com/jQuery.extend/)
	 *
	 */
	Tk.apply = $.extend;

	Tk.apply(Tk, {

		/**
		 * 判断输入值是否"像数组"，像则返回 true，包括 `arguments` `NodeList`都算
		 *
		 * @param {Object} target 
		 * @return {Boolean}
		 * @method
		 */
		isArray: ('isArray' in Array) ? Array.isArray : function(value) {
			return toString.call(value) === '[object Array]';
		}
	});

	for (i in { toString: 1 }) {
		//如果能够遍历到 `toString` 这类特定属性
		//那么以后遍历属性就不需要对这些特定属性作额外遍历
        enumerables = null;
    }

    /**
     * 对于老的浏览器，某些特定的属性无法遍历，这里存放哪些无法遍历的属性名
     * @property {String[]}
     */
    Tk.enumerables = enumerables;

})(window);
